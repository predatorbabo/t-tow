import React, { useMemo, useState, useRef, useEffect } from 'react';
import { Truck, Navigation, SearchX, Plus, Minus, Crosshair, Layers } from 'lucide-react';
import { TruckOwner } from '../types';
 
interface MapProps {
  userLocation: { lat: number; lng: number };
  trucks: TruckOwner[];
  onTruckClick?: (truck: TruckOwner) => void;
  isLoading?: boolean;
  primaryColor?: string;
  mapStyle?: 'normal' | 'roads';
  onMapStyleChange?: (style: 'normal' | 'roads') => void;
}
 
const TILE_SIZE = 256;
 
const MapVisualizer: React.FC<MapProps> = ({ 
  userLocation, 
  trucks = [], 
  onTruckClick, 
  isLoading = false,
  primaryColor = 'blue',
  mapStyle = 'normal',
  onMapStyleChange
}) => {
  // Local view state to allow panning/zooming away from user location
  const [viewState, setViewState] = useState({
    lat: userLocation?.lat || 0,
    lng: userLocation?.lng || 0,
    zoom: 15
  });
 
  const [isDragging, setIsDragging] = useState(false);
  const dragStart = useRef({ x: 0, y: 0 });
  const startCenter = useRef({ lat: 0, lng: 0 });
  const mapRef = useRef<HTMLDivElement>(null);
  const hasInitialized = useRef(false);
 
  // Sync location on load or change, unless user has dragged (optional: could enforce snap back)
  useEffect(() => {
    if (userLocation && (!hasInitialized.current || (viewState.lat === 0 && viewState.lng === 0))) {
      setViewState(prev => ({
        ...prev,
        lat: userLocation.lat,
        lng: userLocation.lng
      }));
      hasInitialized.current = true;
    }
  }, [userLocation]);
 
  // Color mapping
  const colorStyles: Record<string, { bg: string, text: string, border: string, ring: string }> = {
    blue: { bg: 'bg-blue-600', text: 'text-blue-700', border: 'border-blue-200', ring: 'ring-blue-500' },
    indigo: { bg: 'bg-indigo-600', text: 'text-indigo-700', border: 'border-indigo-200', ring: 'ring-indigo-500' },
    rose: { bg: 'bg-rose-600', text: 'text-rose-700', border: 'border-rose-200', ring: 'ring-rose-500' },
    orange: { bg: 'bg-orange-600', text: 'text-orange-700', border: 'border-orange-200', ring: 'ring-orange-500' },
    emerald: { bg: 'bg-emerald-600', text: 'text-emerald-700', border: 'border-emerald-200', ring: 'ring-emerald-500' },
    default: { bg: 'bg-blue-600', text: 'text-blue-700', border: 'border-blue-200', ring: 'ring-blue-500' }
  };
 
  const theme = colorStyles[primaryColor || 'blue'] || colorStyles.default;
 
  // Web Mercator Projection
  const project = (lat: number, lng: number, zoom: number) => {
    let siny = Math.sin((lat * Math.PI) / 180);
    siny = Math.min(Math.max(siny, -0.9999), 0.9999);
    
    return {
      x: TILE_SIZE * (0.5 + lng / 360) * Math.pow(2, zoom),
      y: TILE_SIZE * (0.5 - Math.log((1 + siny) / (1 - siny)) / (4 * Math.PI)) * Math.pow(2, zoom)
    };
  };
 
  const unproject = (x: number, y: number, zoom: number) => {
    const scale = TILE_SIZE * Math.pow(2, zoom);
    const lng = (x / scale - 0.5) * 360;
    const n = 0.5 - y / scale;
    const lat = (360 / Math.PI) * Math.atan(Math.exp(n * 2 * Math.PI)) - 90;
    return { lat, lng };
  };
 
  const centerPixel = useMemo(() => {
    return project(viewState.lat, viewState.lng, viewState.zoom);
  }, [viewState]);
 
  // Calculate tiles to cover the view
  const tiles = useMemo(() => {
    const centerTileX = Math.floor(centerPixel.x / TILE_SIZE);
    const centerTileY = Math.floor(centerPixel.y / TILE_SIZE);
 
    const tileList = [];
    // Load a grid around the center (increase range for smooth panning)
    const range = 2; 
    for (let dx = -range; dx <= range; dx++) {
      for (let dy = -range; dy <= range; dy++) {
        tileList.push({ x: centerTileX + dx, y: centerTileY + dy });
      }
    }
    return tileList;
  }, [centerPixel, viewState.zoom]);
 
  // Handlers
  const handleMouseDown = (e: React.MouseEvent | React.TouchEvent) => {
    setIsDragging(true);
    const clientX = 'touches' in e ? e.touches[0].clientX : (e as React.MouseEvent).clientX;
    const clientY = 'touches' in e ? e.touches[0].clientY : (e as React.MouseEvent).clientY;
    
    dragStart.current = { x: clientX, y: clientY };
    startCenter.current = { lat: viewState.lat, lng: viewState.lng };
  };
 
  const handleMouseMove = (e: React.MouseEvent | React.TouchEvent) => {
    if (!isDragging) return;
    
    const clientX = 'touches' in e ? e.touches[0].clientX : (e as React.MouseEvent).clientX;
    const clientY = 'touches' in e ? e.touches[0].clientY : (e as React.MouseEvent).clientY;
 
    const dx = clientX - dragStart.current.x;
    const dy = clientY - dragStart.current.y;
 
    const startPix = project(startCenter.current.lat, startCenter.current.lng, viewState.zoom);
    const newPix = { x: startPix.x - dx, y: startPix.y - dy };
    const newLatLon = unproject(newPix.x, newPix.y, viewState.zoom);
 
    setViewState(prev => ({ ...prev, lat: newLatLon.lat, lng: newLatLon.lng }));
  };
 
  const handleMouseUp = () => {
    setIsDragging(false);
  };
 
  const handleZoom = (delta: number) => {
    setViewState(prev => ({ ...prev, zoom: Math.min(Math.max(prev.zoom + delta, 3), 19) }));
  };
 
  const handleRecenter = () => {
    if (userLocation) {
      setViewState(prev => ({ ...prev, lat: userLocation.lat, lng: userLocation.lng, zoom: 15 }));
    }
  };
 
  const handleLayerToggle = () => {
    if (onMapStyleChange) {
      onMapStyleChange(mapStyle === 'normal' ? 'roads' : 'normal');
    }
  };
 
  if (isLoading) {
    return (
      <div className="w-full h-64 bg-gray-50 rounded-lg overflow-hidden relative border border-gray-200 shadow-inner isolate">
        <div className="absolute inset-0 grid grid-cols-6 grid-rows-6 opacity-10">
          {[...Array(36)].map((_, i) => (
            <div key={i} className="border-r border-b border-gray-300"></div>
          ))}
        </div>
        <div className="absolute top-1/2 left-1/2 -translate-x-1/2 -translate-y-1/2 w-1/2 h-1/2 border-4 border-gray-200/50 rounded-full animate-pulse"></div>
      </div>
    );
  }
 
  // Ensure integer zoom for tile fetching
  const tileZoom = Math.floor(viewState.zoom);
 
  return (
    <div 
      className="w-full h-64 bg-slate-100 rounded-lg overflow-hidden relative border border-slate-300 shadow-inner group isolate touch-none"
      ref={mapRef}
      onMouseDown={handleMouseDown}
      onMouseMove={handleMouseMove}
      onMouseUp={handleMouseUp}
      onMouseLeave={handleMouseUp}
      onTouchStart={handleMouseDown}
      onTouchMove={handleMouseMove}
      onTouchEnd={handleMouseUp}
      style={{ cursor: isDragging ? 'grabbing' : 'grab' }}
    >
      
      {/* Map Tiles Layer */}
      <div className="absolute inset-0 pointer-events-none bg-[#e1e3e4]">
        {tiles.map((tile) => {
          const tilePixelX = tile.x * TILE_SIZE;
          const tilePixelY = tile.y * TILE_SIZE;
          
          let tileUrl = `https://tile.openstreetmap.org/${tileZoom}/${tile.x}/${tile.y}.png`;
          if (mapStyle === 'roads') {
            // Using CartoDB Positron for a cleaner "Roads" view
            tileUrl = `https://basemaps.cartocdn.com/light_all/${tileZoom}/${tile.x}/${tile.y}.png`;
          }
 
          return (
            <img 
              key={`${tile.x}-${tile.y}-${tileZoom}-${mapStyle}`}
              src={tileUrl}
              alt=""
              className={`absolute max-w-none ${mapStyle === 'roads' ? 'opacity-100 contrast-125' : 'opacity-90 grayscale-[0.2]'}`}
              style={{
                left: `calc(50% + ${tilePixelX - centerPixel.x}px)`,
                top: `calc(50% + ${tilePixelY - centerPixel.y}px)`,
                width: TILE_SIZE,
                height: TILE_SIZE,
              }}
            />
          );
        })}
      </div>
 
      {/* Attribution */}
      <div className="absolute bottom-0 right-0 text-[8px] text-slate-500 bg-white/70 px-1.5 py-0.5 pointer-events-auto z-10">
        {mapStyle === 'roads' ? '© CartoDB' : '© OSM'}
      </div>
 
      {/* Map Controls */}
      <div className="absolute top-2 right-2 flex flex-col space-y-2 z-20 pointer-events-auto">
        <button 
          onClick={() => handleZoom(1)} 
          className="bg-white p-1.5 rounded-md shadow-md text-gray-600 hover:text-blue-600 active:bg-gray-100"
          title="Zoom In"
        >
          <Plus size={16} />
        </button>
        <button 
          onClick={() => handleZoom(-1)} 
          className="bg-white p-1.5 rounded-md shadow-md text-gray-600 hover:text-blue-600 active:bg-gray-100"
          title="Zoom Out"
        >
          <Minus size={16} />
        </button>
        <button 
          onClick={handleRecenter} 
          className="bg-white p-1.5 rounded-md shadow-md text-gray-600 hover:text-green-600 active:bg-gray-100"
          title="Recenter"
        >
          <Crosshair size={16} />
        </button>
        {onMapStyleChange && (
          <button 
            onClick={handleLayerToggle} 
            className={`bg-white p-1.5 rounded-md shadow-md active:bg-gray-100 ${mapStyle === 'roads' ? 'text-blue-600' : 'text-gray-600'}`}
            title="Toggle Layers"
          >
            <Layers size={16} />
          </button>
        )}
      </div>
 
      {/* User Marker */}
      {userLocation && (
        <div 
          className="absolute transform -translate-x-1/2 -translate-y-1/2 z-20 flex flex-col items-center justify-center pointer-events-none transition-all duration-75"
          style={{
            left: `calc(50% + ${project(userLocation.lat, userLocation.lng, viewState.zoom).x - centerPixel.x}px)`,
            top: `calc(50% + ${project(userLocation.lat, userLocation.lng, viewState.zoom).y - centerPixel.y}px)`
          }}
        >
          <div className="relative">
            <div className={`absolute inset-0 rounded-full opacity-30 animate-ping ${theme.bg}`}></div>
            <div className={`border-2 border-white rounded-full p-1.5 shadow-lg relative z-10 ${theme.bg}`}>
              <Navigation size={18} className="text-white transform rotate-0" fill="currentColor" />
            </div>
          </div>
          <span className={`text-[10px] font-bold mt-1 bg-white/90 px-1.5 py-0.5 rounded backdrop-blur-sm shadow-sm ${theme.text}`}>You</span>
        </div>
      )}
 
      {/* Trucks Markers */}
      {trucks.map(truck => {
        if (!truck.location) return null;
        
        const truckPixel = project(truck.location.lat, truck.location.lng, viewState.zoom);
        const offsetX = truckPixel.x - centerPixel.x;
        const offsetY = truckPixel.y - centerPixel.y;
        
        // Optimize rendering: Don't render if far outside viewport
        if (Math.abs(offsetX) > 1000 || Math.abs(offsetY) > 1000) return null;
 
        return (
          <button
            key={truck.id}
            onClick={(e) => {
              e.stopPropagation();
              onTruckClick?.(truck);
            }}
            className="absolute transform -translate-x-1/2 -translate-y-1/2 z-30 group/marker transition-all duration-75 ease-linear hover:z-40"
            style={{ 
              left: `calc(50% + ${offsetX}px)`, 
              top: `calc(50% + ${offsetY}px)` 
            }}
          >
            <div className={`relative p-1.5 rounded-full shadow-md border-2 border-white transition-transform hover:scale-110 ${truck.isAvailable ? 'bg-emerald-500 text-white' : 'bg-gray-400 text-gray-100'}`}>
              <Truck size={16} fill="currentColor" />
            </div>
            
            {/* Label / Tooltip */}
            <div className="absolute bottom-full left-1/2 -translate-x-1/2 mb-1.5 opacity-0 group-hover/marker:opacity-100 transition-opacity pointer-events-none z-50">
              <div className="bg-gray-800 text-white text-[10px] px-2 py-1 rounded shadow-lg whitespace-nowrap">
                {truck.companyName}
              </div>
              <div className="w-2 h-2 bg-gray-800 transform rotate-45 absolute -bottom-1 left-1/2 -translate-x-1/2"></div>
            </div>
          </button>
        );
      })}
 
      {/* Empty State Overlay */}
      {trucks.length === 0 && !isLoading && (
        <div className="absolute inset-0 z-40 bg-white/60 backdrop-blur-[2px] flex flex-col items-center justify-center text-center p-4 pointer-events-none">
          <div className="bg-white p-3 rounded-full shadow-md mb-2">
            <SearchX size={24} className="text-gray-400" />
          </div>
          <p className="text-sm font-bold text-gray-700">No trucks nearby</p>
          <p className="text-xs text-gray-500 mt-1">Try increasing the radius.</p>
        </div>
      )}
    </div>
  );
};
 
export default MapVisualizer;
